//program to find euclidian distance between two points
#include<stdio.h>
#include<math.h>
int main() {
    float x1,x2,y1,y2,d;
    printf("enter the four values");
    scanf("%f%f%f%f",&x1,&x2,&y1,&y2);
    d=sqrt(pow(x2-x1,2)+pow(y2-y1,2));
    printf("the distance is %f",d);
    return 0;
}
